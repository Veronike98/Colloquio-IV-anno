	\begin{defblock}{Cauchy Hypersurface}
		\(S\) is a \emph{Cauchy} hypersurface if it is \emph{achronal}, \emph{spacelike} \& if 
		\(\forall p \notin S\) any inextendible \emph{causal} path through \(p\) intersects \(S\).
	\end{defblock}
